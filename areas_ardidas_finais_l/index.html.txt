<script>
  const map = L.map('map', { zoomControl: true }).setView([39.6, -8.2], 7);

  L.tileLayer('https://{s}.basemaps.cartocdn.com/light_nolabels/{z}/{x}/{y}{r}.png', {
    attribution: '&copy; OpenStreetMap &copy; CARTO',
    maxZoom: 19
  }).addTo(map);

  let municipios = [];
  let municipiosPorNome = {};
  let geojsonLayer = null;
  let selectedLayer = null;

  const estiloNormal = { color: '#6E9A7C', weight: 1, fillColor: '#3D6B4F', fillOpacity: 0.08 };
  const estiloHover = { color: '#3D6B4F', weight: 1.5, fillOpacity: 0.18 };
  const estiloSelecionado = { color: '#C2703D', weight: 2.5, fillColor: '#C2703D', fillOpacity: 0.18 };

  function normalizarNome(txt){
    return txt.toLowerCase().normalize('NFD').replace(/[\u0300-\u036f]/g, '');
  }

  fetch('data/municipios.json')
    .then(r => r.json())
    .then(data => {
      municipios = data;
      municipios.forEach(m => municipiosPorNome[normalizarNome(m.municipio)] = m);
      carregarGeometrias();
    });

  function carregarGeometrias(){
    fetch('data/municipios_geom.geojson')
      .then(r => r.json())
      .then(geojson => {
        geojsonLayer = L.geoJSON(geojson, {
          style: estiloNormal,
          onEachFeature: (feature, layer) => {
            const nome = feature.properties.municipio;
            const chave = normalizarNome(nome);
            const m = municipiosPorNome[chave];
            if(!m) return;

            layer.bindTooltip(nome, { className: 'muni-tooltip', sticky: true });
            layer.on('mouseover', () => layer !== selectedLayer && layer.setStyle(estiloHover));
            layer.on('mouseout', () => layer !== selectedLayer && layer.setStyle(estiloNormal));
            layer.on('click', () => selectMunicipio(m, layer));
          }
        }).addTo(map);
        map.fitBounds(geojsonLayer.getBounds(), { padding: [12, 12] });
      });
  }

  function selectMunicipio(m, layer){
    document.getElementById('empty-state').style.display = 'none';
    const content = document.getElementById('selection-content');
    content.style.display = 'flex';

    document.getElementById('sel-distrito').textContent = m.distrito;
    document.getElementById('sel-municipio').textContent = m.municipio;

    // CAOP
    document.getElementById('link-caop').href = m.caop_zip;
    document.getElementById('size-caop').textContent = `Shapefile (.zip) · ${m.caop_size}`;

    // COS
    document.getElementById('link-cos').href = m.cos_zip;
    document.getElementById('size-cos').textContent = `Shapefile (.zip) · ${m.cos_size}`;

    // OSM Layers
    const osmList = document.getElementById('osm-list');
    osmList.innerHTML = '';
    m.selected_layers.forEach(layerName => {
      const niceName = layerName
        .replace('gis_osm_', '')
        .replace('_free_1', '')
        .replace('_a_free_1', '')
        .replace(/_/g, ' ')
        .replace(/\b\w/g, l => l.toUpperCase());

      const a = document.createElement('a');
      a.className = 'download-item';
      a.innerHTML = `
        <div class="di-left">
          <span class="di-name">${niceName}</span>
          <span class="di-meta">GeoPackage / Shapefile</span>
        </div>
        <span class="di-arrow">&darr;</span>
      `;
      // Aqui podes adicionar o link real quando tiveres os ficheiros
      // a.href = `data/OSM/${m.municipio}/${layerName}.zip`;
      osmList.appendChild(a);
    });

    if(selectedLayer) selectedLayer.setStyle(estiloNormal);
    if(layer){
      layer.setStyle(estiloSelecionado);
      selectedLayer = layer;
      map.fitBounds(layer.getBounds(), { padding: [40, 40], maxZoom: 11 });
    }

    document.getElementById('search-results').classList.remove('active');
  }

  // Pesquisa (mantida igual)
  const searchInput = document.getElementById('search-input');
  const searchResults = document.getElementById('search-results');

  searchInput.addEventListener('input', () => {
    const q = normalizarNome(searchInput.value.trim());
    if(q.length < 1){
      searchResults.classList.remove('active');
      searchResults.innerHTML = '';
      return;
    }

    const matches = municipios.filter(m => normalizarNome(m.municipio).includes(q)).slice(0, 8);

    searchResults.innerHTML = matches.length === 0 
      ? '<div class="search-result-item" style="cursor:default;">Nenhum município encontrado</div>'
      : matches.map(m => `<div class="search-result-item" data-nome="${m.municipio}"><span>${m.municipio}</span><span class="dist">${m.distrito}</span></div>`).join('');

    searchResults.classList.add('active');

    searchResults.querySelectorAll('.search-result-item[data-nome]').forEach(el => {
      el.addEventListener('click', () => {
        const nome = el.dataset.nome;
        const m = municipiosPorNome[normalizarNome(nome)];
        searchInput.value = nome;

        let layerEncontrada = null;
        geojsonLayer.eachLayer(l => {
          if(normalizarNome(l.feature.properties.municipio) === normalizarNome(nome)) layerEncontrada = l;
        });

        selectMunicipio(m, layerEncontrada);
      });
    });
  });

  document.addEventListener('click', e => {
    if(!searchResults.contains(e.target) && e.target !== searchInput) searchResults.classList.remove('active');
  });
</script>
